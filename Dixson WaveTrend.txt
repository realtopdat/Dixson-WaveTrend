// === Signal Slope % (Dix$on angle% style), fixed denominator = 1 ===
f_anglePct_now_fixed(_series) =>
    now  = _series
    prev = _series[1]
    na(now) or na(prev) ? 0.0 :
         (math.atan((now - prev) / 1.0) / (math.pi / 2)) * 100.0

f_intPctStr(_v) =>
    v = nz(_v, 0.0)
    r = math.round(v)
    r == 0 ? "0" : str.tostring(r)

f_hist_get(_arr, _absIdx) =>
    rel = _absIdx - hist_start_idx
    rel >= 0 and rel < array.size(_arr) ? array.get(_arr, rel) : na

f_push_pivot(_posArr, _priceArr, _oscArr, _idx, _price, _osc) =>
    array.unshift(_posArr, _idx)
    array.unshift(_priceArr, _price)
    array.unshift(_oscArr, _osc)
    if array.size(_posArr) > maxarraysize
        array.pop(_posArr)
        array.pop(_priceArr)
        array.pop(_oscArr)

f_sep_ok(_idxA, _priceA, _oscA, _idxB, _priceB, _oscB) =>
    barSep = _idxB - _idxA
    minSepBars = 2
    minRsiSep = minRsiSepInput
    rsiSepOk = minRsiSep <= 0.0 or math.abs(_oscB - _oscA) >= minRsiSep
    barSep >= minSepBars and barSep <= maxbars and rsiSepOk

f_rule_ok(_isLowSide, _isHidden, _priceA, _oscA, _priceB, _oscB) =>
    _isLowSide ? (_isHidden ? (_priceB > _priceA and _oscB < _oscA) : (_priceB < _priceA and _oscB > _oscA)) : (_isHidden ? (_priceB < _priceA and _oscB > _oscA) : (_priceB > _priceA and _oscB < _oscA))

f_path_ok(_idxA, _priceA, _oscA, _idxB, _priceB, _oscB, _isLowSide) =>
    valid = _idxB > _idxA
    if valid and (_idxB - _idxA) > 1
        slopeOsc = (_oscB - _oscA) / (_idxB - _idxA)
        slopePrice = (_priceB - _priceA) / (_idxB - _idxA)
        for i = _idxA + 1 to _idxB - 1
            expOsc = _oscA + slopeOsc * (i - _idxA)
            expPrice = _priceA + slopePrice * (i - _idxA)
            curOsc = f_hist_get(hist_osc, i)
            curPrice = f_hist_get(hist_price, i)
            if na(curOsc) or na(curPrice)
                valid := false
                break
            if _isLowSide
                if curOsc < expOsc or curPrice < expPrice
                    valid := false
                    break
            else
                if curOsc > expOsc or curPrice > expPrice
                    valid := false
                    break
    valid

f_pair_ok(_idxA, _priceA, _oscA, _idxB, _priceB, _oscB, _isLowSide, _isHidden) =>
    sepOk = f_sep_ok(_idxA, _priceA, _oscA, _idxB, _priceB, _oscB)
    ruleOk = f_rule_ok(_isLowSide, _isHidden, _priceA, _oscA, _priceB, _oscB)
    pathOk = f_path_ok(_idxA, _priceA, _oscA, _idxB, _priceB, _oscB, _isLowSide)
    sepOk and ruleOk and pathOk

                    break
    intact

f_pool_ensure_size(_lines, _labels, _aIdx, _bIdx, _scores, _cap) =>
    while array.size(_lines) < _cap
        array.push(_lines, na)
        array.push(_labels, na)
        array.push(_aIdx, na)
        array.push(_bIdx, na)
        array.push(_scores, na)
    while array.size(_lines) > _cap
        ln = array.pop(_lines)
        lb = array.pop(_labels)
        array.pop(_aIdx)
        array.pop(_bIdx)
        array.pop(_scores)
        if not na(ln)
            line.delete(ln)
        if not na(lb)
            label.delete(lb)

f_pool_clear_slot(_lines, _labels, _aIdx, _bIdx, _scores, _slot) =>
    ln = array.get(_lines, _slot)
    lb = array.get(_labels, _slot)
    if not na(ln)
        line.delete(ln)
    if not na(lb)
        label.delete(lb)
    array.set(_lines, _slot, na)
    array.set(_labels, _slot, na)
    array.set(_aIdx, _slot, na)
    array.set(_bIdx, _slot, na)
    array.set(_scores, _slot, na)

f_pool_assign_slot(_lines, _labels, _aIdx, _bIdx, _scores, _slot, _candA, _candB, _candScore, _txt, _col, _isUp) =>
    aOsc = f_hist_get(hist_osc, _candA)
    bOsc = f_hist_get(hist_osc, _candB)
    if not na(aOsc) and not na(bOsc)
        ln = f_upsert_live(array.get(_lines, _slot), _candA, aOsc, _candB, bOsc, _col)
        hidOffset = str.contains(_txt, "H") ? 1.2 : 0.0
        slotOffset = (_isUp ? -1.0 : 1.0) * (_slot * 3.0 + hidOffset)
        labelY = math.min(100.0, math.max(0.0, bOsc + slotOffset))
        labelX = _candB + _slot
        lb = f_upsert_live_label(array.get(_labels, _slot), labelX, labelY, _txt, _col, _isUp)
        array.set(_lines, _slot, ln)
        array.set(_labels, _slot, lb)
        array.set(_aIdx, _slot, _candA)
        array.set(_bIdx, _slot, _candB)
        array.set(_scores, _slot, _candScore)
    else
        f_pool_clear_slot(_lines, _labels, _aIdx, _bIdx, _scores, _slot)

f_candidate_score(_idxA, _priceA, _oscA, _idxB, _priceB, _oscB) =>
    wOsc = 1.0
    wPrc = 0.8
    wLen = 0.25
    lenNorm = math.min(1.0, math.max(0.0, (_idxB - _idxA) / math.max(1.0, maxbars * 1.0)))
    wOsc * math.abs(_oscB - _oscA) + wPrc * math.abs(_priceB - _priceA) + wLen * lenNorm

f_candidate_try_add(_candA, _candB, _candScore, _cap, _candAArr, _candBArr, _candScoreArr) =>
    skip = false
    candSize = array.size(_candAArr)
    if candSize > 0
        for d = 0 to candSize - 1
            exA = array.get(_candAArr, d)
            exB = array.get(_candBArr, d)
            exS = array.get(_candScoreArr, d)
            nearDup = exB == _candB
            if (exA == _candA and exB == _candB) or nearDup
                if _candScore > exS
                    array.set(_candAArr, d, _candA)
                    array.set(_candBArr, d, _candB)
                    array.set(_candScoreArr, d, _candScore)
                skip := true
                break
    if not skip
        if array.size(_candAArr) < _cap
            array.push(_candAArr, _candA)
            array.push(_candBArr, _candB)
            array.push(_candScoreArr, _candScore)
        else
            worstIdx = 0
            worstScore = array.get(_candScoreArr, 0)
            for i = 1 to array.size(_candScoreArr) - 1
                s = array.get(_candScoreArr, i)
                if s < worstScore
                    worstScore := s
                    worstIdx := i
            if _candScore > worstScore
                array.set(_candAArr, worstIdx, _candA)
                array.set(_candBArr, worstIdx, _candB)
                array.set(_candScoreArr, worstIdx, _candScore)

f_top_candidates(_isLowSide, _isHidden, _cap, _posArr, _priceArr, _oscArr, _outA, _outB, _outScore) =>
    array.clear(_outA)
    array.clear(_outB)
    array.clear(_outScore)
    anchorCount = math.min(array.size(_posArr), math.min(array.size(_priceArr), array.size(_oscArr)))
    maxAnch = math.min(liveMaxAnchors, anchorCount)
    checks = 0
    if maxAnch > 0
        for ai = 0 to maxAnch - 1
            aIdx = array.get(_posArr, ai)
            aPrice = array.get(_priceArr, ai)
            aOsc = array.get(_oscArr, ai)
            liveFreshStart = math.max(0, bar_index - pp)
            bStart = math.max(aIdx + 2, liveFreshStart)
            bEnd = math.min(bar_index, aIdx + maxbars)
            if bStart <= bEnd
                perAnchorFound = 0
                runExtreme = _isLowSide ? 10e10 : -10e10
                for b = bStart to bEnd
                    if perAnchorFound >= liveMaxBPerAnchor or checks >= liveScanBudget
                        break
                    bPrice = f_hist_get(hist_price, b)
                    bOsc = f_hist_get(hist_osc, b)
                    if not na(bPrice) and not na(bOsc)
                        isExtreme = _isLowSide ? (bPrice < runExtreme) : (bPrice > runExtreme)
                        if isExtreme
                            runExtreme := bPrice
                            checks += 1
                            rulePass = f_rule_ok(_isLowSide, _isHidden, aPrice, aOsc, bPrice, bOsc)
                            if rulePass
                                okPair = f_pair_ok(aIdx, aPrice, aOsc, b, bPrice, bOsc, _isLowSide, _isHidden)
                                if okPair and f_live_b_intact(_isLowSide, b, bPrice)
                                    sc = f_candidate_score(aIdx, aPrice, aOsc, b, bPrice, bOsc)
                                    f_candidate_try_add(aIdx, b, sc, _cap, _outA, _outB, _outScore)
                                    perAnchorFound += 1
                if perAnchorFound < liveMaxBPerAnchor and checks < liveScanBudget
                    for b = bStart to bEnd
                        if perAnchorFound >= liveMaxBPerAnchor or checks >= liveScanBudget
                            break
                        bPrice = f_hist_get(hist_price, b)
                        bOsc = f_hist_get(hist_osc, b)
                        if not na(bPrice) and not na(bOsc)
                            checks += 1
                            rulePass = f_rule_ok(_isLowSide, _isHidden, aPrice, aOsc, bPrice, bOsc)
                            if rulePass
                                okPair = f_pair_ok(aIdx, aPrice, aOsc, b, bPrice, bOsc, _isLowSide, _isHidden)
                                if okPair and f_live_b_intact(_isLowSide, b, bPrice)
                                    sc = f_candidate_score(aIdx, aPrice, aOsc, b, bPrice, bOsc)
                                    f_candidate_try_add(aIdx, b, sc, _cap, _outA, _outB, _outScore)
                                    perAnchorFound += 1

f_pool_update_hybrid(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, _candA, _candB, _candScore, _cap, _txt, _col, _isUp, _isLowSide, _enabled) =>
    f_pool_ensure_size(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, _cap)
    candCount = math.min(array.size(_candA), math.min(array.size(_candB), array.size(_candScore)))
    if not _enabled
        for i = 0 to _cap - 1
            f_pool_clear_slot(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, i)
    else
        kept = array.new_bool(0)
        assigned = array.new_bool(0)
        candUsed = array.new_bool(0)
        for i = 0 to _cap - 1
            array.push(kept, false)
            array.push(assigned, false)
        if candCount > 0
            for c = 0 to candCount - 1
                array.push(candUsed, false)

        for s = 0 to _cap - 1
            curA = array.get(_aIdxArr, s)
            curB = array.get(_bIdxArr, s)
            if not na(curA) and not na(curB) and candCount > 0
                curBPrice = f_hist_get(hist_price, curB)
                slotIntact = not na(curBPrice) and f_live_b_intact(_isLowSide, curB, curBPrice)
                if slotIntact
                    for c = 0 to candCount - 1
                        if not array.get(candUsed, c)
                            ca = array.get(_candA, c)
                            cb = array.get(_candB, c)
                            if ca == curA and cb == curB
                                array.set(kept, s, true)
                                array.set(assigned, s, true)
                                array.set(candUsed, c, true)
                                sc = array.get(_candScore, c)
                                f_pool_assign_slot(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, s, ca, cb, sc, _txt, _col, _isUp)
                                break

        if candCount > 0
            for c = 0 to candCount - 1
                if not array.get(candUsed, c)
                    int target = na
                    for s = 0 to _cap - 1
                        if na(array.get(_aIdxArr, s)) and not array.get(assigned, s)
                            target := s
                            break
                    if na(target)
                        int worstSlot = na
                        worstScore = 10e10
                        for s = 0 to _cap - 1
                            if not array.get(kept, s)
                                slotScore = array.get(_scoreArr, s)
                                slotScore := na(slotScore) ? -10e10 : slotScore
                                if na(worstSlot) or slotScore < worstScore
                                    worstScore := slotScore
                                    worstSlot := s
                        target := worstSlot
                    if not na(target)
                        ca = array.get(_candA, c)
                        cb = array.get(_candB, c)
                        sc = array.get(_candScore, c)
                        f_pool_assign_slot(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, target, ca, cb, sc, _txt, _col, _isUp)
                        array.set(assigned, target, true)
                        array.set(candUsed, c, true)

        for s = 0 to _cap - 1
            if not array.get(assigned, s)
                f_pool_clear_slot(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, s)

f_pool_remove_match(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, _a, _b) =>
    found = false
    for s = 0 to array.size(_aIdxArr) - 1
        sa = array.get(_aIdxArr, s)
        sb = array.get(_bIdxArr, s)
        if not na(sa) and not na(sb) and sa == _a and sb == _b
            f_pool_clear_slot(_lines, _labels, _aIdxArr, _bIdxArr, _scoreArr, s)
            found := true
            break
    found

f_pc_live_sync(_srcA, _srcB, _dstLines, _dstLabels, _cap, _txt, _col, _isUp, _enabled) =>
    while array.size(_dstLines) < _cap
        array.push(_dstLines, na)
        array.push(_dstLabels, na)
    while array.size(_dstLines) > _cap
        lnPop = array.pop(_dstLines)
        lbPop = array.pop(_dstLabels)
        if not na(lnPop)
            line.delete(lnPop)
        if not na(lbPop)
            label.delete(lbPop)
    for i = 0 to _cap - 1
        ln = array.get(_dstLines, i)
        lb = array.get(_dstLabels, i)
        valid = _enabled and i < array.size(_srcA) and i < array.size(_srcB)
        aIdx = valid ? array.get(_srcA, i) : na
        bIdx = valid ? array.get(_srcB, i) : na
        aPrice = valid and not na(aIdx) ? f_hist_get(hist_price, aIdx) : na
        bPrice = valid and not na(bIdx) ? f_hist_get(hist_price, bIdx) : na
        if valid and not na(aIdx) and not na(bIdx) and not na(aPrice) and not na(bPrice)
            if na(ln)
                ln := line.new(aIdx, aPrice, bIdx, bPrice, xloc=xloc.bar_index, extend=extend.none, color=color.new(_col, liveTransp), width=pc_width, style=line.style_dashed, force_overlay=true)
            else
                line.set_xy1(ln, aIdx, aPrice)
                line.set_xy2(ln, bIdx, bPrice)
                line.set_style(ln, line.style_dashed)
                line.set_width(ln, pc_width)
                line.set_color(ln, color.new(_col, liveTransp))
            if pc_show_labels
                if na(lb)
                    lb := label.new(bIdx, bPrice, text=_txt, xloc=xloc.bar_index, yloc=yloc.price, style=_isUp ? label.style_label_up : label.style_label_down, size=pc_lbl_size, color=color.new(_col, liveTransp), textcolor=color.white, force_overlay=true)
                else
                    label.set_x(lb, bIdx)
                    label.set_y(lb, bPrice)
                    label.set_text(lb, _txt)
                    label.set_style(lb, _isUp ? label.style_label_up : label.style_label_down)
                    label.set_size(lb, pc_lbl_size)
                    label.set_color(lb, color.new(_col, liveTransp))
                    label.set_textcolor(lb, color.white)
            else
                if not na(lb)
                    label.delete(lb)
                    lb := na
        else
            if not na(ln)
                line.delete(ln)
                ln := na
            if not na(lb)
                label.delete(lb)
                lb := na
        array.set(_dstLines, i, ln)
        array.set(_dstLabels, i, lb)
